```
git clone git@github.com:wierton/nscscc-system-18 --recursive
cd nscscc-system-18 && make run
```

or 

```
git clone git@github.com:wierton/nscscc-system-18
git submodule update --init --recursive
cd nscscc-system-18 && make run
```
